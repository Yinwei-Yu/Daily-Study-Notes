# 1. 代码修改后的快速重建
./dev_build.sh go           # 只修改了Go代码
./dev_build.sh cpp          # 只修改了C++代码
./dev_build.sh proto        # 修改了proto定义

# 2. 运行测试
./dev_build.sh test-unit proxy    # 测试特定组件
./dev_build.sh test-go            # 测试所有Go代码
./dev_build.sh test-cpp           # 测试所有C++代码

# 3. 完整构建（首次或大修改后）
./dev_build.sh full --no-azure

# 4. 清理重来
./dev_build.sh clean
./dev_build.sh deps
./dev_build.sh full
